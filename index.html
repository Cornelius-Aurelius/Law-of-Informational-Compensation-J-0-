Verification Method

import numpy as np

p = np.random.rand(1000); p /= p.sum()
q = np.ones(1000) / 1000

def kl(p, q): 
    return np.sum(p * np.log(p / q))

for i in range(1000):
    d = kl(p,q)
    if d < 0.01:
        print("Equilibrium reached:", i)
        break
